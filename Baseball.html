    group_by(yearID) %>%
    summarize(avg_weight = mean(avg_weight, na.rm = TRUE)) %>%
    ggplot(aes(x = yearID, y = avg_weight)) +
    geom_point() +
    geom_line() +
    labs(x = "Year", y = "Average Weight", title = "Average Player Weight by Year")
